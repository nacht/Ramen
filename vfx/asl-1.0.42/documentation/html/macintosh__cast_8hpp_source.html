00083 template <typename R>
00084 inline R cf_cast(::CFTypeRef x)
00085 {
00086     if (x == 0)
00087         return reinterpret_cast<R>(0);
00088 
00089     ::CFTypeID src_type_id(::CFGetTypeID(x));
00090     ::CFTypeID dest_type_id(implementation::cftypeid_proc_type<R>()());
00091 
00092     return reinterpret_cast<R>(src_type_id == dest_type_id ? x : 0);
00093 }
