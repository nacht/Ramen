00312 template <typename NewFormat>
00313 void callback(std::ios_base::event ev, std::ios_base& strm, int idx)
00314     {
00315     if (ev == std::ios_base::erase_event)
00316         {
00317         try
00318             {
00319             delete static_cast<NewFormat*>(strm.pword(idx));
00320 
00321             strm.pword(idx) = 0;
00322             }
00323         catch (...)
00324             { }
00325         }
00326     else if (ev == std::ios_base::copyfmt_event)
00327         {
00328         NewFormat* old(static_cast<NewFormat*>(strm.pword(idx)));
00329
